<script>

/* ====== Datos del juego (edítalos) ====== */

const DOCS = [

  {

    id:"doc1",

    title:"Foto del laboratorio — Acto 1",

    ref:"ATLAS‑03/DOC‑1",

    intro:"Imagen encontrada tras revisar la agenda de Víctor. Ajusta brillo/contraste para revelar detalles.",

    img:"A_photograph_captures_the_interior_of_a_dimly_lit_.png", // coloca aquí la ruta real

    auto:{b:1.35,c:1.6},

    locked:false

  },

  {

    id:"doc2",

    title:"Sección 2 (bloqueada)",

    ref:"ATLAS‑03/DOC‑2",

    intro:"Contenido protegido. Requiere la clave descubierta en el Acto 1.",

    img:"placeholder_locked.jpg", // opcional: una imagen borrosa/candado

    auto:{b:1,c:1},

    locked:true,

    pass:"Carmen1706" // ← Clave de ejemplo

  },

  {

    id:"doc3",

    title:"Documento final (bloqueado)",

    ref:"ATLAS‑03/DOC‑3",

    intro:"Acceso restringido. Necesitas una segunda clave del Acto 2.",

    img:"placeholder_locked.jpg",

    auto:{b:1,c:1},

    locked:true,

    pass:"MUELLE" // ← Clave de ejemplo

  }

];


/* ====== Estado y utilidades ====== */

const $ = sel => document.querySelector(sel);

const stateKey = "atlas_web_state_v1";

const loadState = ()=> JSON.parse(localStorage.getItem(stateKey) || "{}");

const saveState = s => localStorage.setItem(stateKey, JSON.stringify(s));


let state = loadState();

if(!state.unlocked) state.unlocked = { doc1:true }; // el doc1 está abierto por defecto

if(!state.notes) state.notes = {};

if(!state.current) state.current = "doc1";


/* ====== Render lista lateral ====== */

const docList = $("#docList");

function renderList(){

  docList.innerHTML = "";

  DOCS.forEach(d=>{

    const li = document.createElement("button");

    li.className = "doc";

    li.setAttribute("role","listitem");

    li.dataset.id = d.id;

    const name = document.createElement("span");

    name.className="name";

    name.textContent = d.title;

    const lock = document.createElement("span");

    lock.className="lock";

    const isOpen = state.unlocked[d.id];

    lock.textContent = d.locked && !isOpen ? "🔒" : "🔓";

    li.appendChild(name); li.appendChild(lock);

    if(state.current===d.id) li.setAttribute("aria-current","true");

    li.onclick = ()=> openDoc(d.id);

    docList.appendChild(li);

  });

}


/* ====== Abrir documento ====== */

function openDoc(id){

  state.current = id; saveState(state);

  const d = DOCS.find(x=>x.id===id);

  const isOpen = state.unlocked[id] || !d.locked;


  $("#refId").textContent = d.ref;

  $("#docTitle").textContent = d.title;

  $("#docIntro").textContent = d.intro;


  // reset filtros

  $("#br").value = 1; $("#ct").value = 1; setFilters();


  if(isOpen){

    $("#docImage").src = d.img;

    $("#docImage").alt = d.title;

  } else {

    $("#docImage").src = d.img || "";

    $("#docImage").alt = "Documento bloqueado";

    // lanzar modal de desbloqueo

    unlockDialog.showModal();

  }


  // Notas

  $("#notes").value = state.notes[id] || "";


  renderList(); // refrescar estado de candados

}


/* ====== Filtros ====== */

const setFilters = ()=>{

  const img = $("#docImage");

  img.style.setProperty("--b", $("#br").value);

  img.style.setProperty("--c", $("#ct").value);

};

$("#br").oninput = $("#ct").oninput = setFilters;

$("#reveal").onclick = ()=>{

  const d = DOCS.find(x=>x.id===state.current);

  $("#br").value = d.auto.b; $("#ct").value = d.auto.c; setFilters();

};

$("#reset").onclick = ()=>{ $("#br").value = 1; $("#ct").value = 1; setFilters(); };

$("#download").onclick = ()=>{

  const d = DOCS.find(x=>x.id===state.current);

  if(!d) return;

  const a = document.createElement("a"); a.href = d.img; a.download = d.img.split("/").pop();

  a.click();

};


/* ====== Notas persistentes ====== */

$("#notes").addEventListener("input", e=>{

  state.notes[state.current] = e.target.value; saveState(state);

});


/* ====== Navegación ====== */

$("#prev").onclick = ()=>{

  const idx = DOCS.findIndex(x=>x.id===state.current);

  if(idx>0) openDoc(DOCS[idx-1].id);

};

$("#next").onclick = ()=>{

  const idx = DOCS.findIndex(x=>x.id===state.current);

  if(idx<DOCS.length-1) openDoc(DOCS[idx+1].id);

};


/* ====== Desbloqueo ====== */

const unlockDialog = $("#unlockDialog");

$("#unlockBtn").onclick = (ev)=>{

  ev.preventDefault();

  const code = $("#unlockInput").value.trim();

  const d = DOCS.find(x=>x.id===state.current);

  if((d.pass||"").toLowerCase() === code.toLowerCase()){

    state.unlocked[d.id] = true; saveState(state);

    $("#unlockMsg").textContent = "Desbloqueado ✅";

    setTimeout(()=>{ unlockDialog.close(); openDoc(d.id); }, 300);

  } else {

    $("#unlockMsg").textContent = "Clave incorrecta.";

  }

};

unlockDialog.addEventListener("close", ()=>{ $("#unlockInput").value=""; $("#unlockMsg").textContent=""; });


/* ====== Inicial ====== */

renderList();

openDoc(state.current);


// Doble toque sobre la imagen = auto‑ajuste (móvil)

let t=0; $("#docImage").addEventListener('touchend', ()=>{ const now=Date.now(); if(now-t<350){ $("#reveal").click(); } t=now;});

</script>
